programa ejercicio1
procesos
  proceso limpiarFloresEsquina
  comenzar
    mientras (HayFlorEnLaEsquina)
      tomarFlor
  fin
  
  proceso limparPapelesEsquina
  comenzar
    mientras (HayPapelEnLaEsquina)
      tomarPapel
  fin

areas
  AreaCompartida: AreaC (10,10,10,10)
  AreaFloreros: AreaPC (1,1,5,10)
  AreaPapeleros: AreaPC (6,1,10,9)
  AreaFlorero1: AreaP (6,10,6,10)
  AreaFlorero2: AreaP (7,10,7,10)
  AreaPapelero1: AreaP (8,10,8,10)
  AreaPapelero2: AreaP (9,10,9,10)
robots
  robot recolectorFlores
  variables
    avActual: numero
    caActual: numero
    avenida: numero
    calle: numero
  comenzar
    repetir 5
      avActual:= PosAv
      caActual:= PosCa
      Random(avenida, 1,5)
      Random(calle, 1,10)
      BloquearEsquina(avenida, calle)
      Pos(avenida, calle)
      limpiarFloresEsquina
      Pos(avActual, caActual)
      LiberarEsquina(avenida, calle)
    BloquearEsquina(10,10)
    Pos(10,10)
    mientras (HayFlorEnLaBolsa)
      depositarFlor
    Pos(avActual, caActual)
    LiberarEsquina(10,10)
  fin
  
  robot recolectorPapeles
  variables
    avActual: numero
    caActual: numero
    avenida: numero
    calle: numero
  comenzar
    repetir 3
      avActual:= PosAv
      caActual:= PosCa
      Random(avenida, 6,10)
      Random(calle, 1,9)
      BloquearEsquina(avenida, calle)
      Pos(avenida, calle)
      limpiarFloresEsquina
      Pos(avActual, caActual)
      LiberarEsquina(avenida, calle)
    BloquearEsquina(10,10)
    Pos(10,10)
    mientras (HayPapelEnLaBolsa)
      depositarPapel
    Pos(avActual, caActual)
    LiberarEsquina(10,10)
  fin
  
variables
  florero1: recolectorFlores
  florero2: recolectorFlores
  papelero1: recolectorPapeles
  papelero2: recolectorPapeles
comenzar
  AsignarArea(florero1, AreaFloreros)
  AsignarArea(florero2, AreaFloreros)
  AsignarArea(papelero1, AreaPapeleros)
  AsignarArea(papelero2, AreaPapeleros)
  
  AsignarArea(florero1, AreaFlorero1)
  AsignarArea(florero2, AreaFlorero2)
  AsignarArea(papelero1, AreaPapelero1)
  AsignarArea(papelero2, AreaPapelero2)
  
  AsignarArea(florero1, AreaCompartida)
  AsignarArea(florero2, AreaCompartida)
  AsignarArea(papelero1, AreaCompartida)
  AsignarArea(papelero2, AreaCompartida)
  
  Iniciar(florero1, 6,10)
  Iniciar(florero2, 7,10)
  Iniciar(papelero1, 8,10)
  Iniciar(papelero2, 9,10)
fin
