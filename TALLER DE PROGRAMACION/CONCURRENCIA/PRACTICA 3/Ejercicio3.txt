  proceso limpiarFlorEsq(E numAv: numero; E numCa: numero; ES cantFlores: numero)
  comenzar
    Pos(numAv, numCa)
    si (HayFlorEnLaEsquina)
      tomarFlor
      cantFlores:= cantFlores + 1
  fin

robots
  robot limpiador
  variables
    av: numero
    ca: numero
    avActual: numero
    caActual: numero
    cantFlores: numero
  comenzar
    cantFlores:= 0
    RecibirMensaje(av,jefe)
    RecibirMensaje(ca,jefe)
    avActual:= PosAv
    caActual:= PosCa
    BloquearEsquina(av,ca)
    limpiarFlorEsq(av,ca, cantFlores)
    Pos(avActual,caActual)
    LiberarEsquina(av,ca)
    EnviarMensaje(cantFlores, jefe)
  fin
  
  robot jefe
  variables
    avenida: numero
    calle: numero
    Flores1: numero
    Flores2: numero
    Flores3: numero
    Flores4: numero
    max: numero
    robotMax: numero
  comenzar
    Random(avenida, 2,10)
    Random(calle, 2,10)
    EnviarMensaje(avenida,r1)
    EnviarMensaje(calle,r1)
    EnviarMensaje(avenida,r2)
    EnviarMensaje(calle,r2)
    EnviarMensaje(avenida,r3)
    EnviarMensaje(calle,r3)
    EnviarMensaje(avenida,r4)
    EnviarMensaje(calle,r4)
    RecibirMensaje(Flores1, r1)
    RecibirMensaje(Flores2, r2)
    RecibirMensaje(Flores3, r3)
    RecibirMensaje(Flores4, r4)
    si (Flores1 > max)
      max:= Flores1
      robotMax:= 1
    si (Flores2 > max)
      max:= Flores2
      robotMax:= 2
    si (Flores3 > max)
      max:= Flores3
      robotMax:= 3
    si (Flores4 > max)
      max:= Flores3
      robotMax:= 4
    Informar(robotMax)
  fin
