# INCISO A

programa Ejercicio1.A

procesos
  proceso tomarFloresEsquina (ES numCa: numero; ES cantFlores: numero)
  comenzar
    mientras (HayFlorEnLaEsquina)
      tomarFlor
      cantFlores:= cantFlores + 1
      numCa:= numCa + 1
  fin
  
  proceso depositarTotalFlores (E cantFlores: numero)
  comenzar
    repetir cantFlores
      depositarFlor
  fin
  
  proceso recorrerAvenida(ES numAv: numero; ES numCa: numero; ES cantFlores: numero; ES cantSinFlor: numero)
  comenzar
    cantFlores:= 0
    Pos(numAv, numCa)
    repetir 99
      si (HayFlorEnLaEsquina)
        tomarFloresEsquina(numCa, cantFlores)
      sino
        cantSinFlor:= cantSinFlor + 1
      mover
    si (HayFlorEnLaEsquina)
      tomarFloresEsquina(numCa, cantFlores)
    sino
      cantSinFlor:= cantSinFlor + 1
    tomarFloresEsquina(numCa, cantFlores)
    depositarTotalFlores(cantFlores)
  fin
  
areas
  ciudad: AreaC (1,1,100,100)
  
robots

  robot robot1
  variables
    numCa: numero
    numAv: numero
    cantFlores: numero
    cantSinFlor: numero
  comenzar
    numCa:= 1
    cantFlores:= 0
    cantSinFlor:= 0
    
    { AVENIDA 1 }
    numAv:= 1
    recorrerAvenida(numAv, numCa, cantFlores, cantSinFlor)
    Informar(cantFlores)
    Informar(cantSinFlor)
    
    { AVENIDA 3 }
    numAv:= 3
    cantSinFlor:= 0
    recorrerAvenida(numAv, numCa, cantFlores, cantSinFlor)
    Informar(cantFlores)
    Informar(cantSinFlor)
    
    { AVENIDA 5 }
    numAv:= 5
    cantSinFlor:= 0
    recorrerAvenida(numAv, numCa, cantFlores, cantSinFlor)
    Informar(cantFlores)
    Informar(cantSinFlor)
  fin
  
  
variables
  r1: robot1
comenzar
  AsignarArea(r1, ciudad)
  Iniciar(r1, 1,1)
fin
