procesos
  proceso tomarFloresEsquina (ES numCa: numero; ES cantFlores: numero)
  comenzar
    mientras (HayFlorEnLaEsquina)
      tomarFlor
      cantFlores:= cantFlores + 1
      numCa:= numCa + 1
  fin
  
  proceso depositarTotalFlores (E cantFlores: numero)
  comenzar
    repetir cantFlores
      depositarFlor
  fin
  
  proceso recorrerAvenidaUno(ES numCa: numero; ES cantFlores: numero; ES cantSinFlor: numero)
  comenzar
    cantFlores:= 0
    Pos(1, numCa)
    repetir 99
      si (HayFlorEnLaEsquina)
        tomarFloresEsquina(numCa, cantFlores)
      sino
        cantSinFlor:= cantSinFlor + 1
      mover
    si (HayFlorEnLaEsquina)
      tomarFloresEsquina(numCa, cantFlores)
    sino
      cantSinFlor:= cantSinFlor + 1
    tomarFloresEsquina(numCa, cantFlores)
    depositarTotalFlores(cantFlores)
  fin

  robot robot1
  variables
    numCa: numero
    cantFlores: numero
    cantSinFlor: numero

    cantSinFlor:= 0

  fin
