  proceso hacerEscalon(ES medidaEscalon: numero; ES cantEscalones: numero)
  variables
    auxFlores: numero
    auxPapeles: numero
  comenzar
    auxFlores:= 0
    auxPapeles:= 0
    repetir 2
      repetir medidaEscalon
        mientras (HayPapelEnLaEsquina)
          tomarPapel
          auxPapeles:= auxPapeles + 1
        mientras (HayFlorEnLaEsquina)
          tomarFlor
          auxFlores:= auxFlores + 1
        mientras (HayPapelEnLaBolsa)
          depositarPapel
        mientras (HayFlorEnLaBolsa)
          depositarFlor
        mover
      derecha
    mientras (HayPapelEnLaEsquina)
      tomarPapel
      auxPapeles:= auxPapeles + 1
    mientras (HayFlorEnLaEsquina)
      tomarFlor
      auxFlores:= auxFlores + 1
    mientras (HayPapelEnLaBolsa)
      depositarPapel
    mientras (HayFlorEnLaBolsa)
      depositarFlor
    repetir 2
      derecha
    si (auxPapeles - auxFlores = 1)
      cantEscalones:= cantEscalones + 1
    medidaEscalon:= medidaEscalon + 1
  fin
