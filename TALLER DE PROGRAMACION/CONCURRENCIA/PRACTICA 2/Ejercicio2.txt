  proceso hacerEscalera (E alto: numero; ES cantMax: numero)
  variables
    cantPapeles: numero
    cantFlores: numero
  comenzar 
    repetir 4
      cantPapeles:= 0
      cantFlores:= 0
      repetir alto
        mientras (HayFlorEnLaEsquina)
          tomarFlor
          cantFlores:= cantFlores + 1
        mientras (HayPapelEnLaEsquina)
          tomarPapel
          cantPapeles:= cantPapeles + 1
        mover
      derecha
      mientras (HayFlorEnLaEsquina)
        tomarFlor
        cantFlores:= cantFlores + 1
      mientras (HayPapelEnLaEsquina)
        tomarPapel
        cantPapeles:= cantPapeles + 1
      mover
      si (cantFlores > cantPapeles)
        cantMax:= cantMax + 1
      repetir 3
        derecha
  fin
