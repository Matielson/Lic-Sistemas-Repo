  proceso recorrerYJuntarPapeles(ES cantPapeles: numero)
  comenzar
    repetir 9
      mientras (HayPapelEnLaEsquina)
        tomarPapel
        cantPapeles:= cantPapeles + 1
      mover
    mientras (HayPapelEnLaEsquina)
      tomarPapel

  fin
